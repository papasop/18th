\title{\textbf{The Universal Completion Theorem:\\[4pt]
Non-Collapsing Variational Systems and the Necessity\\[2pt]
of Basin-Independent Access}}
\author{Y.\,Y.\,N.\ Li}
\date{}

\begin{document}
\maketitle

\begin{abstract}
What structure must exist in a variational system that forbids the annihilation of
all configurations?  We show: exactly one---a smooth generative manifold~$M$, universally
accessible, permanently retentive, and equipped with cost-free degrees of freedom.
The argument rests on two axioms: \emph{realizability} (the system admits a structurally
stable, nontrivial extremum that does not collapse to the trivial state) and
\emph{continuous symmetry} (the cost functional is invariant under a connected Lie group
of positive dimension acting freely on nontrivial configurations).
Continuous symmetry implies that every critical point is non-isolated.
Structural stability then forces the Hessian to be indefinite---yielding
Lorentzian signature on every two-dimensional subspace containing directions
of both signs---\emph{without any assumption on the sign of a temporal direction}: the
spectral-boundary argument excludes both the semidefinite cases $H\ge 0$ and $H\le 0$.
From these two axioms, twelve structural properties of~$M$ are derived: access is
basin-independent, capture depends on alignment alone, gauge directions are
indestructible under any symmetry-preserving extension, and exclusion is
geometrically self-imposed.
By \emph{necessity} we mean that these properties are forced by the axioms:
any system satisfying realizability and continuous symmetry must exhibit them.
\end{abstract}

%% ====================================================================
\section{The Axioms}\label{sec:axioms}
%% ====================================================================

The entire paper follows from two axioms.

Let $\mathcal{F}$ be a real separable Hilbert space with inner product
$\langle\cdot,\cdot\rangle$ and norm $\|\cdot\|$.
The variational calculus on~$\mathcal{F}$ follows Palais~\cite{Palais}.

\begin{definition}[Cost functional]\label{def:cost}
A \emph{cost functional} is a twice Fr\'echet-differentiable map
$S:\mathcal{F}\to\mathbb{R}_{\ge 0}$ satisfying:
\begin{itemize}
\item $S[\psi]=0$ if and only if $\psi=0$;
\item at every critical point $\psi^*$ ($\nabla S[\psi^*]=0$), the Hessian
  $H[\psi^*]$ is a Fredholm operator (finite-dimensional kernel, closed range).
\end{itemize}
\end{definition}

\begin{definition}[Functional gradient]\label{def:grad}
$\nabla S[\psi]\in\mathcal{F}$ is the Riesz representative:
$\delta S[\psi;\delta\psi]=\langle\delta\psi,\nabla S[\psi]\rangle$.
\end{definition}

\begin{definition}[Hessian, null space]\label{def:Hess}
$H[\psi]v=\frac{d}{d\varepsilon}\big|_{\varepsilon=0}\nabla S[\psi+\varepsilon v]$,
bounded and self-adjoint.
$\ker H[\psi]=\{v\in\mathcal{F}:H[\psi]v=0\}$.
\end{definition}

\begin{definition}[Generative manifold]\label{def:M}
\[
M=\bigl\{\psi\in\mathcal{F}:\nabla S[\psi]=0,\;\dim\ker H[\psi]>0\bigr\}.
\]
\end{definition}

\begin{definition}[Completion channel]\label{def:channel}
A \emph{completion channel} is a map $C:\mathcal{F}\to\mathcal{F}$ representing
external coupling.  The dynamics is
$\partial_t\psi=-\nabla S[\psi]+\varepsilon\,C[\psi]$,
where $\varepsilon>0$ is the coupling strength.
The gradient flow $-\nabla S$ drives states toward $\psi=0$;
the channel~$C$ provides an alternative direction.
\end{definition}

\begin{remark}[Physical realisations of~$C$]
In open quantum systems, $C$ corresponds to the Lindblad dissipator.
In classical field theory, $C$ corresponds to external source terms.
The theorems require only the \emph{existence} of some~$C$ with
nonzero kernel projection, not a specific choice.
\end{remark}

\begin{axiom}[Realizability]\label{ax:R}
$S$ admits an attainable, nontrivial extremum $\psi^*\neq 0$ satisfying:
\begin{itemize}
\item[\textup{(i)}] \textbf{Stationarity:} $\nabla S[\psi^*]=0$.
\item[\textup{(ii)}] \textbf{Non-collapse:} No nearby trajectory of
  $\partial_t\psi=-\nabla S[\psi]$ converges to $\psi\equiv 0$.
\item[\textup{(iii)}] \textbf{Structural stability:} \textup{(i)--(ii)}, the non-isolation
  of~$\psi^*$ in $\{\nabla S=0\}$, and the value $\dim\ker H[\psi^*]$
  all persist under all sufficiently small perturbations of~$S$
  (including perturbations that break gauge symmetry).
\item[\textup{(iv)}] \textbf{Singular separation:}
  The closure of $\{\psi:\nabla S[\psi]=0,\;\dim\ker H[\psi]>0\}$
  does not intersect the singular set
  $\Sigma_S=\{\psi:S\text{ is not twice differentiable at }\psi\}$.
\end{itemize}
\end{axiom}

If $S\ge 0$ with $S=0\Leftrightarrow\psi\equiv 0$, gradient flow drives every
nontrivial state to annihilation.  This is \emph{Euclidean collapse}.
Axiom~\ref{ax:R} excludes it.

\begin{remark}[Content of each condition in Axiom~\ref{ax:R}(iii)]
Axiom~\ref{ax:R}(iii) packages three separately used robustness conditions:
\begin{itemize}
\item \emph{Stationarity and non-collapse persist} ---
  used in the universal-access proof (Theorem~\ref{thm:univ}).
\item \emph{Non-isolation persists} ---
  used in the indefiniteness proof (Theorem~\ref{thm:indef}): the
  perturbation~$S_\varepsilon$ preserves stationarity and non-collapse
  but destroys non-isolation, yielding a contradiction only because
  non-isolation is required to persist.
\item \emph{$\dim\ker H[\psi^*]$ persists} ---
  used in the spectral-gap proof (Proposition~\ref{prop:gap}):
  if an eigenvalue could be pushed to zero, $\dim\ker H$ would increase,
  contradicting kernel-dimension stability.
  This is the strongest of the three conditions; it implies non-isolation
  persistence but not conversely.
\end{itemize}
All three are consequences of the standard notion of structural stability
(qualitative equivalence under perturbation) applied to the critical set of~$S$.
\end{remark}

\begin{axiom}[Continuous symmetry]\label{ax:G}
There exists a connected Lie group~$G$ with $\dim G\ge 1$ acting smoothly
on~$\mathcal{F}$ such that:
\begin{itemize}
\item[\textup{(i)}] \textbf{Invariance:} $S[g\cdot\psi]=S[\psi]$ for all
  $g\in G$, $\psi\in\mathcal{F}$.
\item[\textup{(ii)}] \textbf{Free action on nontrivial configurations:}
  $\psi\neq 0$ and $g\cdot\psi=\psi$ imply $g=e$.
\end{itemize}
\end{axiom}

\begin{remark}[Physical content of Axiom~\ref{ax:G}]
The invariance condition~(i) is satisfied by every known fundamental interaction:
$\mathrm{U}(1)$, $\mathrm{SU}(2)$, $\mathrm{SU}(3)$, $\mathrm{Diff}(\mathcal{M})$.
The free-action condition~(ii) restricts the configuration space to the
\emph{irreducible sector}: in gauge field theory, reducible connections have
nontrivial stabilisers and must be excluded.
This is a standard restriction in gauge theory
(cf.~the irreducible stratum in~\cite{AB}).
The axiom does not postulate any property of the critical set;
the consequences for the critical set are derived below.
\end{remark}

\begin{remark}[Consistency of the axioms: an explicit model]\label{rem:example}
Let $\Sigma$ be a compact Riemannian manifold with $\dim\Sigma\ge 1$, and let
$\mathcal{F}=H^{1}(\Sigma,\mathbb{C})$.  Let $G=\mathrm{U}(1)$ act by constant
phase rotation: $g\cdot\psi=e^{i\theta}\psi$.  Define
\[
S[\psi]=\int_{\Sigma}\bigl(|\nabla\psi|^{2}+|\psi|^{4}(|\psi|^{2}-1)^{2}
+\varepsilon\,|\psi|^{4}\bigr)\,d\mu,\qquad\varepsilon>0\;\text{small}.
\]
Then $S\ge 0$ and $S[\psi]=0\Leftrightarrow\psi=0$
(since $|\psi|^{4}[(|\psi|^{2}-1)^{2}+\varepsilon]=0$ only when $|\psi|=0$).
For small~$\varepsilon$, there is a $\mathrm{U}(1)$-orbit of approximately constant
critical points with $|\psi^{*}|\approx 1/\!\sqrt{2}$, satisfying
Axiom~\ref{ax:R}(i).
The Hessian at~$\psi^{*}$ is a Schr\"odinger operator $-\Delta+W$ on~$\Sigma$,
which is Fredholm.  In the radial direction, the constant mode ($k=0$) has
eigenvalue $W=V''(s_{0})<0$ (since $s_{0}\approx 1/2$ is a local maximum
of the potential restricted to constant configurations), while high-frequency
modes have eigenvalue $\lambda_{k}+W>0$ for $k$ large.
The Hessian is therefore indefinite for any compact~$\Sigma$---consistent
with Theorem~\ref{thm:indef}.
The transverse spectral gap can be verified independently:
$-\Delta+W$ on compact~$\Sigma$ has discrete spectrum, and the gap between
zero (the gauge eigenvalue) and the nearest nonzero eigenvalue is bounded
below by $\min(|W|,\lambda_{1}(\Sigma))>0$, where $\lambda_{1}$ is the first
positive eigenvalue of~$-\Delta$.
This confirms Axiom~\ref{ax:R}(iii) (including $\dim\ker H$ persistence)
without circularity.
Axiom~\ref{ax:R}(ii) and~(iv) and Axiom~\ref{ax:G} can be verified by standard
elliptic-regularity arguments.
\end{remark}

\section{Global Non-Isolation from Symmetry}\label{sec:GNI}
%% ====================================================================

\begin{proposition}[Continuous symmetry implies global non-isolation]\label{prop:GNI}
Under Axiom~\textup{\ref{ax:G}}, if $\nabla S[\psi^*]=0$ and $\psi^*\neq 0$, the orbit
$G\cdot\psi^*$ is a submanifold of $\{\nabla S=0\}$ with
$\dim(G\cdot\psi^*)=\dim G\ge 1$.
In particular, $\dim\ker H[\psi^*]\ge\dim G\ge 1$.
\end{proposition}

\begin{proof}
For any $\xi\in\mathfrak{g}$, Axiom~\ref{ax:G}(i) gives
$S[\exp(t\xi)\cdot\psi^*]=S[\psi^*]$ for all~$t$.
For any $g\in G$, differentiating $S[g\cdot\psi]=S[\psi]$ with respect to~$\psi$
at $\psi=g^{-1}\cdot\psi^*$ gives
$\langle\nabla S[\psi^*],\,Dg\cdot v\rangle=\langle\nabla S[g^{-1}\cdot\psi^*],\,v\rangle$
for all~$v$; since $\nabla S[\psi^*]=0$, the left side vanishes, and since $Dg$ is
invertible, $\nabla S[g^{-1}\cdot\psi^*]=0$.
Hence $\nabla S[g\cdot\psi^*]=0$ for every $g\in G$, and
$G\cdot\psi^*\subset\{\nabla S=0\}$.
The action is free (Axiom~\ref{ax:G}(ii)), so the orbit map is an immersion:
$\dim(G\cdot\psi^*)=\dim G\ge 1$.

Differentiating $\nabla S[\exp(t\xi)\cdot\psi^*]=0$ at $t=0$ gives
$H[\psi^*]\xi_\mathcal{F}(\psi^*)=0$
for all $\xi\in\mathfrak{g}$, where
$\xi_\mathcal{F}(\psi^*)=\frac{d}{dt}\big|_{t=0}\exp(t\xi)\cdot\psi^*$
is the infinitesimal generator at~$\psi^*$.
(Here we use $\nabla S[g\cdot\psi^*]=0$ for all $g$, not merely $S[g\cdot\psi^*]=\mathrm{const}$.)
Since the action is free, $\xi\mapsto\xi_\mathcal{F}(\psi^*)$ is injective,
so $\ker H[\psi^*]$ contains a subspace of dimension $\ge\dim G$.
\end{proof}

\begin{remark}[Comparison with Noether's theorem]
Noether's theorem~\cite{Noether} derives conserved quantities from continuous symmetry.
Proposition~\ref{prop:GNI} derives non-isolation of critical structure from the
same premise.  Both follow from group invariance of the action functional,
though Noether's result is established physics while the present application
is new.
\end{remark}

\section{Algebraic Consequences}\label{sec:alg}
%% ====================================================================

\begin{theorem}[Indefiniteness without temporal cost assumption]\label{thm:indef}
Under Axioms~\textup{\ref{ax:R}} and~\textup{\ref{ax:G}}, $H[\psi^*]$ is indefinite.
No assumption on the sign of $\langle e_t,He_t\rangle$ for any direction~$e_t$
is required.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:GNI}, $d:=\dim\ker H[\psi^*]\ge 1$.
Let $\{v_1,\ldots,v_d\}$ be an orthonormal basis of $\ker H[\psi^*]$.

\textbf{Case~1:} Suppose $H[\psi^*]\ge 0$ (positive semidefinite).
Then zero is the infimum of the spectrum.
Define $S_\varepsilon[\psi]:=S[\psi]+\frac{\varepsilon}{2}\sum_{i=1}^d
\langle\psi-\psi^*,v_i\rangle^2$ for $\varepsilon>0$.
Since the perturbation is non-negative, $S_\varepsilon\ge S\ge 0$.
At~$\psi^*$: $\nabla S_\varepsilon[\psi^*]=0$.
The perturbed Hessian is $H_\varepsilon=H+\varepsilon\,P_{\ker H}$, where
$P_{\ker H}=\sum_{i=1}^d\langle v_i,\cdot\rangle v_i$ is the orthogonal projection
onto $\ker H$.  Then
$H_\varepsilon v_i=\varepsilon\,v_i$.  Hence $\ker H_\varepsilon=\{0\}$:
$\psi^*$ is isolated under~$S_\varepsilon$, contradicting Axiom~\ref{ax:R}(iii).

\textbf{Case~2:} Suppose $H[\psi^*]\le 0$ (negative semidefinite).
Then zero is the supremum of the spectrum.
The \emph{same} perturbation $S_\varepsilon$ gives $H_\varepsilon=H+\varepsilon\,P_{\ker H}$.
On $\ker H$: eigenvalues become $+\varepsilon>0$.
On $(\ker H)^\perp$: eigenvalues of $H$ are strictly negative.
Hence $\ker H_\varepsilon=\{0\}$: $\psi^*$ is again isolated, contradicting
Axiom~\ref{ax:R}(iii).

In both cases, semidefiniteness places the zero eigenvalues at the spectral
boundary, where the perturbation eliminates them.
The only remaining possibility is that $H[\psi^*]$ has both positive and
negative eigenvalues: $H[\psi^*]$ is indefinite.
\end{proof}

\begin{remark}[The perturbation breaks gauge symmetry]
$S_\varepsilon$ is not $G$-invariant: $g$ moves $\psi$ but the centre~$\psi^*$ and
directions~$v_i$ are fixed.  This is why Axiom~\ref{ax:R}(iii) requires
persistence under perturbations that break gauge symmetry.
\end{remark}

\begin{remark}[Elimination of the temporal cost condition]
In previous realizability-based derivations, indefiniteness required a positive
temporal cost condition as an independent assumption.
Theorem~\ref{thm:indef} eliminates it: the spectral-boundary argument applies
to \emph{both} $H\ge 0$ and $H\le 0$, exhausting the semidefinite cases without
needing the sign of any particular direction.
\end{remark}

\begin{theorem}[Lorentzian signature]\label{thm:Lor}
Under Axioms~\textup{\ref{ax:R}} and~\textup{\ref{ax:G}}, there exist $e_+,e_-\in\mathcal{F}$
with $\langle e_+,H[\psi^*]e_+\rangle>0$ and $\langle e_-,H[\psi^*]e_-\rangle<0$.
On $\mathrm{span}\{e_+,e_-\}$ the signature is $(1,1)$: Lorentzian.
The null cone $\mathcal{L}=\{\delta\psi:\langle\delta\psi,H[\psi^*]\delta\psi\rangle=0\}$
defines causal structure.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:indef} gives both positive and negative eigenvalues.
The restriction to any two-dimensional subspace containing one of each has
signature $(1,1)$.
\end{proof}

\begin{proposition}[Morse--Bott regularity]\label{prop:MB}
Under Axioms~\textup{\ref{ax:R}} and~\textup{\ref{ax:G}},
$\dim\ker H[\psi]$ is locally constant on $\{\nabla S=0\}$ near~$\psi^*$.
The critical set is a smooth (Morse--Bott) submanifold
(cf.~\cite{Morse,Bott}).
\end{proposition}

\begin{proof}
Suppose $\dim\ker H[\psi]$ is not locally constant near~$\psi^*$:
there exists a sequence $\psi_n\to\psi^*$ in $\{\nabla S=0\}$ with
$\dim\ker H[\psi_n]\neq\dim\ker H[\psi^*]=:d^*$.
Let $\{v_1,\ldots,v_{d^*}\}$ be an orthonormal basis of $\ker H[\psi^*]$
and let $\eta$ be a smooth cutoff with $\eta(r)=1$ for $r\le 1$, $\eta(r)=0$
for $r\ge 2$.  For small $\varepsilon,\rho>0$, define
\[
S_\varepsilon[\psi]:=S[\psi]+\frac{\varepsilon}{2}\sum_{i=1}^{d^*}
\langle\psi-\psi^*,v_i\rangle^2\;\eta\!\!\left(\frac{\|\psi-\psi^*\|}{\rho}\right).
\]
At $\psi^*$, $\nabla S_\varepsilon[\psi^*]=0$ (since
$\langle\psi^*-\psi^*,v_i\rangle=0$).
But near the dimension-jump locus the perturbation generically resolves the
singularity: branches with $\dim\ker H\neq d^*$ are split away.
After perturbation, the topology of the critical set near~$\psi^*$ has changed
under an arbitrarily small perturbation, contradicting Axiom~\ref{ax:R}(iii).
By the constant-rank theorem, the critical set is smooth.
\end{proof}

\begin{remark}[Precedent in gauge theory]
The Morse--Bott condition for a gauge-invariant functional was first exploited
by Atiyah and Bott~\cite{AB} in Yang--Mills theory.  Here the property is
derived from structural stability, not assumed.
\end{remark}

\begin{proposition}[Transverse spectral gap]\label{prop:gap}
Under Axioms~\textup{\ref{ax:R}} and~\textup{\ref{ax:G}},
together with the Fredholm condition (Definition~\textup{\ref{def:cost}}),
there exists $\lambda>0$
such that $|\langle v,H[\psi^*]v\rangle|\ge\lambda\|v\|^2$ for all
$v\perp\ker H[\psi^*]$.
\end{proposition}

\begin{proof}
Since $H[\psi^*]$ is Fredholm (Definition~\ref{def:cost}), its non-zero spectrum
is discrete.  Write $\sigma_+(H)=\{\mu\in\sigma(H):\mu>0\}$ and
$\sigma_-(H)=\{\mu\in\sigma(H):\mu<0\}$ for the positive and negative parts
of the spectrum.
If $\inf\sigma_+(H)=0$, there exists a sequence of positive
eigenvalues $\lambda_n\to 0$.  By analytic perturbation theory for
self-adjoint Fredholm operators~\cite{Kato}, a perturbation of
size $O(\lambda_n)$ pushes $\lambda_n$ to zero, increasing
$\dim\ker H$---contradicting
Axiom~\ref{ax:R}(iii), which requires $\dim\ker H[\psi^*]$ to be stable
under small perturbations.  Hence $\inf\sigma_+(H)>0$.
The same argument on $-H$ gives $\sup\sigma_-(H)<0$.
Setting $\lambda=\min(\inf\sigma_+,|\sup\sigma_-|)$ gives the gap.
\end{proof}

\section{Uniqueness and Universal Access}\label{sec:unique}
%% ====================================================================

\begin{theorem}[Uniqueness of the non-collapsing completion]\label{thm:unique}
Under Axioms~\textup{\ref{ax:R}} and~\textup{\ref{ax:G}},
$M$ is the unique non-collapsing completion of~$\mathcal{F}$.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:GNI}, every nontrivial critical point has
$\dim\ker H\ge\dim G\ge 1$, hence lies in~$M$.
Therefore $\{\nabla S=0\}\setminus\{0\}\subset M$.
Any non-collapsing completion $M'$ satisfies $M'\subset\{\nabla S=0\}\setminus\{0\}$
(since $\psi=0$ is the collapsed state), hence $M'\subset M$.
\end{proof}

\begin{theorem}[Universal completion is necessary]\label{thm:univ}
For every open set $U\subset\mathcal{F}$, there exists $\psi\in U$ with
$\|P_{\ker H[\psi]}(C[\psi])\|>0$.
No open region is excluded from access to~$M$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:unique}, $M$ is the unique non-collapsing completion.
Suppose $P_{\ker H[\psi]}(C[\psi])=0$ for all $\psi\in U$.
Then capture into~$M$ from~$U$ is impossible.
Since $M$ is unique, configurations in~$U$ have no non-collapsing destination:
gradient flow drives them toward $\psi=0$, violating Axiom~\ref{ax:R}(ii).
No such~$U$ can exist.
\end{proof}

\begin{remark}[Scope of the proof]
The proof uses Axiom~\ref{ax:R}(ii) as a global condition:
no nontrivial configuration is condemned to collapse.
As stated, Axiom~\ref{ax:R}(ii) is local (nearby~$\psi^*$).
The global reach follows from two additional properties of gradient flow:
$S$ is a Lyapunov function ($\dot{S}=-\|\nabla S\|^{2}\le 0$),
and the only critical points are $M\cup\{0\}$ (Theorem~\ref{thm:unique}).
Together these ensure that every trajectory eventually approaches $M\cup\{0\}$,
at which point the local condition applies.
\end{remark}

\begin{theorem}[Indestructibility of gauge directions]\label{thm:interlock}
Let $S_e=S+R$ be an extension that preserves Axiom~\textup{\ref{ax:G}}:
$S_e[g\cdot\psi]=S_e[\psi]$ for all $g\in G$.
Then at every nontrivial critical point of~$S_e$,
\[
\ker H_e[\psi^*]\supseteq\{\xi_\mathcal{F}(\psi^*):\xi\in\mathfrak{g}\}.
\]
No gauge-preserving extension can close a gauge direction.
\end{theorem}

\begin{proof}
$S_e$ satisfies Axiom~\ref{ax:G} with the same~$G$.
By Proposition~\ref{prop:GNI} applied to~$S_e$:
$H_e[\psi^*]\xi_\mathcal{F}(\psi^*)=0$ for all $\xi\in\mathfrak{g}$.
Hence $\{\xi_\mathcal{F}(\psi^*):\xi\in\mathfrak{g}\}\subset\ker H_e[\psi^*]$.
\end{proof}

\begin{remark}[Scope of indestructibility]
Directions in $\ker H[\psi^*]$ beyond the gauge orbit (if any) are not
protected by this theorem.  The gauge directions---which include all
connecting directions along Class~B paths within gauge orbits---are the
ones that cannot be closed.
\end{remark}

\section{Persistence and Capture}\label{sec:persist}
%% ====================================================================

\begin{theorem}[Persistence without continuous injection]\label{thm:persist}
Under pure gradient flow ($\varepsilon=0$):
if $\psi(t_0)\in M$ then $\psi(t)\in M$ for all $t\ge t_0$.
If $C$~is removed at time~$t_0$ after $\psi(t_0)\in M$, no retrogression occurs.
A single moment of alignment suffices permanently once the channel is removed.
\end{theorem}

\begin{proof}
On~$M$, $\nabla S[\psi]=0$, so the gradient flow $\partial_t\psi=-\nabla S[\psi]$
is identically zero: $M$ is invariant.
If $\psi(t_0)\in M$, then $\psi(t)=\psi(t_0)\in M$ for all $t\ge t_0$.
Removing~$C$ at $t_0$ restores pure gradient flow, and invariance applies.
\end{proof}

\begin{remark}[Invariance versus stability]
Persistence is an invariance result, not a stability result.
Since $H[\psi^*]$ is indefinite, $M$ is a saddle of the gradient flow:
perturbations in the $V_+$~direction (positive eigenvalues of~$H$) are restored,
but perturbations in the $V_-$~direction (negative eigenvalues) grow
exponentially under gradient flow.
Reaching~$M$ requires the channel~$C$; once on~$M$, remaining on~$M$
requires only the absence of transverse forcing.
\end{remark}

\begin{remark}[Role of singular separation]
Axiom~\ref{ax:R}(iv) ensures $\overline{M}\cap\Sigma_S=\varnothing$:
no trajectory on~$M$ can reach the singular set of~$S$.
Persistence therefore extends to all future time without encountering
points where the Hessian ceases to exist.
\end{remark}

\begin{remark}[Two classes of motion on~$M$]
\emph{Class~A} (transverse): fluctuations perpendicular to~$M$.
The spectral gap (Proposition~\ref{prop:gap}) ensures $|\langle v,Hv\rangle|\ge\lambda\|v\|^2$
for $v\perp\ker H$.  Positive-eigenvalue directions are restored by gradient flow;
negative-eigenvalue directions are repelled (see the Remark above).
\emph{Class~B} (free): tangential motion along $\ker H$, zero cost,
unconstrained---$\dim G$ directions of cost-free motion at each point,
and $M$ itself may be non-compact, so Class~B trajectories extend without bound.
\end{remark}

\begin{lemma}[Structure of~$M$]\label{lem:Mstr}
$\dim\ker H[\psi]$ is locally constant on~$M$ (Proposition~\ref{prop:MB}).
Hence $M$ is a smooth submanifold with $T_\psi M\supseteq\ker H[\psi]$, and
$\frac{d^2}{d\varepsilon^2}\big|_{\varepsilon=0}S[\psi+\varepsilon\delta\psi]=0$
for $\delta\psi\in T_\psi M$.
\end{lemma}

\begin{proof}
Immediate from Proposition~\ref{prop:MB} and the constant-rank theorem.
\end{proof}

\begin{theorem}[Scale invariance of capture]\label{thm:scale}
$\|P_{\ker H[\psi]}(C[\psi])\|>0$ is invariant under $C\mapsto\lambda C$
for all $\lambda>0$.  Capture depends on alignment, not magnitude.
\end{theorem}

\begin{proof}
$P_{\ker H}(\lambda C)=\lambda\,P_{\ker H}(C)$.
\end{proof}

\begin{remark}[Physical content of scale invariance]
The path to~$M$ requires no force, only direction.
\end{remark}

\begin{definition}[Uniform kernel overlap]\label{def:UKO}
$C$ has \emph{uniform kernel overlap} on~$U$ if
$\|P_{\ker H[\psi]}(C[\psi])\|/\|C[\psi]\|\ge\delta^*>0$
for all $\psi\in U$ with $C[\psi]\neq 0$.
\end{definition}

\begin{definition}[Kernel-projected coherence]\label{def:KPC}
Fix $\psi^\star\in M$ and a unit vector $u^\star\in\ker H[\psi^\star]$.
The coherence of~$C$ at~$\psi$ is
$\kappa(\psi;C):=|\langle u^\star,P_{\ker H[\psi]}(C[\psi])\rangle|/\|P_{\ker H[\psi]}(C[\psi])\|$
when $P_{\ker H[\psi]}(C[\psi])\neq 0$, and $\kappa=0$ otherwise.
\end{definition}

\begin{theorem}[Phase-locking]\label{thm:PL}
Under (i)~transverse spectral gap (Proposition~\ref{prop:gap}),
(ii)~neutrality along $\ker H$,
(iii)~uniform kernel overlap on~$U$,
(iv)~$\dot{S}\le-\sigma<0$ outside~$U$:
if $\kappa(\psi_0;C)\ge\kappa^*>0$, there exists finite~$\tau_{\mathrm{lock}}$
after which the kernel-projected coherence satisfies $\kappa\ge\kappa^*/2$.
\end{theorem}

\begin{proof}[Proof outline]
Condition~(iv) forces $S$ to decrease at rate~$\sigma$ outside~$U$, so the
trajectory enters~$U$ in time $\le S[\psi_0]/\sigma$.  Inside~$U$,
condition~(iii) guarantees kernel overlap $\ge\delta^*$, and
the spectral gap suppresses transverse drift.  Coherence is bounded
below by continuity, giving phase-locking after finite $\tau_{\mathrm{lock}}$.
Quantitative estimates are deferred to future work.
\end{proof}

\begin{corollary}[Global capture chain]\label{cor:chain}
Under condition~(iv) of Theorem~\ref{thm:PL} globally,
$\psi_0\xrightarrow{\text{energy descent}}U\xrightarrow{\text{phase-locking}}M$
for any $\psi_0\in\mathcal{F}$.
\end{corollary}

\begin{proof}
Condition~(iv) gives energy descent into~$U$;
Theorem~\ref{thm:PL} gives phase-locking within~$U$;
Theorem~\ref{thm:univ} ensures no open region is excluded.
\end{proof}

\begin{corollary}[Cost-independence of access]\label{cor:costindep}
Access to~$M$ is independent of the initial cost $S[\psi_0]$.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:univ}, every open set has access to~$M$.
For any $c>0$, the sublevel set $\{\psi:S[\psi]<c\}$ and the superlevel set
$\{\psi:S[\psi]>c\}$ are both open (by continuity of~$S$), hence both have access.
\end{proof}

\begin{definition}[Kernel-denial]\label{def:KD}
$\psi$ exhibits \emph{kernel-denial} if $P_{\ker H[\psi]}(C[\psi])=0$.
\end{definition}

\begin{lemma}[Exclusion is self-imposed]\label{lem:excl}
If kernel-denial persists along a trajectory, the trajectory is excluded from~$M$.
Such persistence is impossible on any open set.
\end{lemma}

\begin{proof}
If $P_{\ker H[\psi(t)]}(C[\psi(t)])=0$ for all large~$t$, no kernel alignment
occurs and the trajectory cannot approach~$M$ via~$C$.
By Theorem~\ref{thm:univ}, $P_{\ker H}(C)\neq 0$ somewhere in every open set,
so kernel-denial cannot hold on any open set.
\end{proof}

\section{Connectedness and Internal Structure}\label{sec:connect}
%% ====================================================================

\begin{proposition}[Kernel-connected completions merge]\label{prop:merge}
If $\psi_1,\psi_2\in M$ are connected by a $C^1$~path $\gamma$ with
constant cost, kernel tangency ($\dot\gamma(s)\in\ker H[\gamma(s)]$),
and $\dot\gamma\neq 0$, then $\gamma\subset M$.
\end{proposition}

\begin{proof}
$H[\gamma(s)]\dot\gamma(s)=0$ gives
$\frac{d}{ds}\nabla S[\gamma(s)]=0$.
Since $\nabla S[\gamma(0)]=0$, stationarity propagates.
Since $\dot\gamma\neq 0$ and $\dot\gamma\in\ker H$,
every point has $\dim\ker H>0$, hence $\gamma\subset M$.
\end{proof}

\begin{corollary}[Gauge connecting directions are indestructible]\label{cor:connect}
If $v=\xi_\mathcal{F}(\psi)$ for some $\xi\in\mathfrak{g}$ is tangent to a Class~B
path connecting two components of~$M$, then $v\in\ker H_e[\psi]$ for every
gauge-preserving extension~$S_e$.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:interlock}, gauge directions lie in $\ker H_e$.
\end{proof}

\section{Summary}\label{sec:summary}
%% ====================================================================

\subsection*{Core geometry of~$M$ (from Axioms~\ref{ax:R} and~\ref{ax:G} alone)}

These properties concern the cost functional~$S$ and the manifold~$M$.
They do not require a completion channel~$C$.

\begin{enumerate}
\item \textbf{Uniqueness} (Thm~\ref{thm:unique}):
  every non-collapsing completion is contained in~$M$.
\item \textbf{Indefiniteness} (Thm~\ref{thm:indef}):
  $H[\psi^*]$ has both positive and negative eigenvalues.
\item \textbf{Lorentzian signature} (Thm~\ref{thm:Lor}):
  every two-dimensional subspace containing directions of both signs has
  signature $(1,1)$.
\item \textbf{Morse--Bott regularity} (Prop~\ref{prop:MB}):
  $M$ is smooth; $\dim\ker H$ is locally constant.
\item \textbf{Spectral gap} (Prop~\ref{prop:gap}):
  $|\langle v,Hv\rangle|\ge\lambda\|v\|^2$ for $v\perp\ker H$.
\item \textbf{Indestructible gauge directions} (Thm~\ref{thm:interlock}):
  no gauge-preserving extension can close a gauge direction.
\item \textbf{Kernel-connected merging} (Prop~\ref{prop:merge}):
  components linked by zero-cost paths are parts of the same~$M$.
\item \textbf{Persistence} (Thm~\ref{thm:persist}):
  $M$ is invariant under gradient flow; arrival is permanent.
\end{enumerate}

\subsection*{Access properties (additionally require a completion channel~$C$)}

These properties concern the interaction between~$C$ and~$M$.
They are consequences of universal access (Theorem~\ref{thm:univ}),
which itself requires the existence of~$C$ with nonzero kernel projection.

\begin{enumerate}\setcounter{enumi}{8}
\item \textbf{Universal access} (Thm~\ref{thm:univ}):
  for every open set, $P_{\ker H}(C)\neq 0$ somewhere.
\item \textbf{Scale-invariant capture} (Thm~\ref{thm:scale}):
  capture depends on alignment, not magnitude.
\item \textbf{Self-imposed exclusion} (Lem~\ref{lem:excl}):
  kernel-denial cannot persist on any open set.
\item \textbf{Global capture chain} (Cor~\ref{cor:chain}):
  under additional energy-descent conditions,
  $\psi_0\to U\to M$ from any initial state.
\end{enumerate}

\begin{remark}[What the separation shows]
The eight geometric properties (1--8) are unconditional consequences of
the axioms; they hold even in closed systems with no external coupling.
The four access properties (9--12) require a completion channel~$C\neq 0$.
In a closed system ($C=0$), $M$ still exists with all its geometric structure,
but no mechanism steers configurations toward it.
\end{remark}

\begin{remark}[Relation to Noether's theorem]
Noether's theorem~\cite{Noether} derives conserved quantities from continuous symmetry.
The present work derives a different structural consequence---the existence of a
unique non-collapsing manifold with indestructible gauge directions---from
the same premise, augmented by structural stability.
The analogy is limited: Noether's theorem is established physics with a century
of verification, while the present framework awaits independent examination.
\end{remark}
