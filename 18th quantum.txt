\title{\textbf{Unitarity, Gauge Symmetry, and the Necessity\\[4pt]
of Non-Collapsing Completion}}
\author{Y.\,Y.\,N.\ Li}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Let $\mathcal{F}$ be a real Hilbert space equipped with a non-negative cost functional
$S:\mathcal{F}\to\mathbb{R}_{\ge 0}$ vanishing only at the origin, a one-parameter
unitary group $U(t)$ preserving the norm, and a continuous gauge group~$G$ acting freely
on nontrivial configurations with $S$ invariant under~$G$.
We show that (i)~unitarity forbids collapse to the trivial state;
(ii)~gauge invariance forces every nontrivial critical point of~$S$ to have a
positive-dimensional kernel in its Hessian; (iii)~structural stability of the critical
set forces the Hessian to be indefinite (yielding Lorentzian signature) \emph{without any
assumption on the sign of a temporal direction}---the indefiniteness follows purely from
the spectral-boundary argument applied to both the semidefinite cases $H\ge 0$ and
$H\le 0$.  The critical set with nontrivial kernel forms a unique smooth Morse--Bott
manifold~$M$, and no open region of~$\mathcal{F}$ is excluded from access to~$M$.
Twelve structural properties of~$M$ are derived from the three premises alone.
\end{abstract}

%% ====================================================================
\section{Framework}\label{sec:framework}
%% ====================================================================

\subsection*{The configuration space and the cost functional}

Let $\mathcal{F}$ be a real separable Hilbert space with inner product
$\langle\cdot,\cdot\rangle$ and norm $\|\cdot\|$.  Elements $\psi\in\mathcal{F}$
are \emph{configurations}.  The zero vector $\psi=0$ represents the trivial
(non-existent) state.  The variational calculus on~$\mathcal{F}$ follows
Palais~\cite{Palais}.

\begin{definition}[Cost functional]\label{def:cost}
A \emph{cost functional} is a twice Fr\'echet-differentiable map
$S:\mathcal{F}\to\mathbb{R}_{\ge 0}$ satisfying $S[\psi]=0$ if and only if
$\psi=0$.
\end{definition}

\begin{definition}[Gradient, Hessian, generative manifold]\label{def:derived}
\mbox{}
\begin{itemize}
\item $\nabla S[\psi]\in\mathcal{F}$: the Riesz representative of the first variation.
\item $H[\psi]v:=\frac{d}{d\varepsilon}\big|_{\varepsilon=0}\nabla S[\psi+\varepsilon v]$:
  the Hessian, bounded and self-adjoint.
\item $\ker H[\psi]:=\{v\in\mathcal{F}:H[\psi]v=0\}$.
\item $M:=\bigl\{\psi\in\mathcal{F}:\nabla S[\psi]=0,\;\dim\ker H[\psi]>0\bigr\}$:
  the \emph{generative manifold}.
\end{itemize}
\end{definition}

\subsection*{Two dynamics on $\mathcal{F}$}

Two dynamics coexist on~$\mathcal{F}$.  Their distinction is essential.

\begin{itemize}
\item \textbf{Collapse flow} (hypothetical):
  $\partial_t\psi=-\nabla S[\psi]$.
  This gradient flow decreases~$S$ monotonically and drives every nontrivial
  state toward $\psi=0$.  If this were the only dynamics, all structure
  would be annihilated.
\item \textbf{Physical dynamics} (actual):
  $\psi(t)=U(t)\psi(0)$, where $U(t)$ is a strongly continuous one-parameter
  group of unitary operators on~$\mathcal{F}$.
  This evolution preserves $\|\psi\|$ and is incompatible with collapse.
\end{itemize}

The cost functional~$S$ is a \emph{structural} object: it defines which states
are critical ($\nabla S=0$), which directions are cost-free ($\ker H$), and
which quadratic form governs the local geometry ($H$).
The unitary group $U(t)$ is a \emph{dynamical} object: it determines how
states evolve in time.
The interplay between the two---structure from~$S$, dynamics from~$U(t)$---is
the subject of this paper.

\subsection*{Completion channel}

\begin{definition}[Completion channel]\label{def:channel}
A \emph{completion channel} is a map $C:\mathcal{F}\to\mathcal{F}$ representing
external coupling or influence.  The full dynamics is
$\partial_t\psi=A[\psi]+\varepsilon\,C[\psi]$,
where $A[\psi]$ generates the unitary evolution and $\varepsilon>0$ is the
coupling strength.  The channel $C$ is not derived from the premises; it is
part of the framework.
\end{definition}

\section{Three Premises}\label{sec:premises}
%% ====================================================================

\begin{description}
\item[Premise U (Unitarity).]
  There exists a strongly continuous one-parameter unitary group
  $U(t)$ on~$\mathcal{F}$ governing the physical dynamics:
  $\psi(t)=U(t)\psi(0)$ with $\|U(t)\psi\|=\|\psi\|$ for all~$t$.

\item[Premise G (Gauge symmetry).]
  There exists a connected Lie group~$G$ with $\dim G\ge 1$ acting smoothly
  on~$\mathcal{F}$ such that:
  (i)~$S[g\cdot\psi]=S[\psi]$ for all $g\in G$, $\psi\in\mathcal{F}$;
  (ii)~the action is free on nontrivial configurations:
  $\psi\neq 0$ and $g\cdot\psi=\psi$ imply $g=e$.

\item[Premise S (Structural stability).]
  The critical structure of~$S$---specifically, the property that nontrivial
  critical points are non-isolated and that $\dim\ker H$ is locally
  constant on the critical set---persists under all sufficiently small
  perturbations of~$S$ (including perturbations that break gauge symmetry).
\end{description}

\begin{remark}[Empirical status]
Premise~U is the foundational postulate of quantum mechanics.
Premise~G is satisfied by every known fundamental interaction:
$\mathrm{U}(1)$, $\mathrm{SU}(2)$, $\mathrm{SU}(3)$, $\mathrm{Diff}(\mathcal{M})$.
Premise~S is the assertion that physical law is robust: the qualitative
structure of the critical set does not depend on the precise form of~$S$.
This is a standard genericity assumption in dynamical systems and is satisfied
by every known physical theory.
\end{remark}

\begin{remark}[What is not assumed]
No Hamiltonian $\hat{H}$ appears as a premise.  In particular, the positive
temporal cost condition ($\langle e_t,He_t\rangle>0$ for some direction~$e_t$),
which is typically imposed as an independent assumption in realizability-based derivations of Lorentzian signature,
is \emph{derived} here (Theorem~\ref{thm:indef}) from the spectral-boundary
argument.  The number of independent inputs is reduced from three to two
physical premises plus one genericity condition.
\end{remark}

%% ====================================================================
\section{Derivation of the Structural Conditions}\label{sec:derive}
%% ====================================================================

\begin{proposition}[Non-collapse from unitarity]\label{prop:NC}
Under Premise~U, no nontrivial state evolves to the trivial state:
if $\psi(0)\neq 0$, then $\psi(t)\neq 0$ for all~$t$.
\end{proposition}

\begin{proof}
$\|\psi(t)\|=\|U(t)\psi(0)\|=\|\psi(0)\|>0$.
Since $S[\psi]=0$ if and only if $\psi=0$, and $\psi(t)\neq 0$,
the trivial state is never reached.
\end{proof}

\begin{remark}[Scope of non-collapse]
Non-collapse is a statement about the physical (unitary) dynamics, not about
the gradient flow.  The gradient flow \emph{would} collapse every state to
$\psi=0$; unitarity prevents this.  The cost functional~$S$ defines the
\emph{landscape} of persistence; unitarity determines that this landscape
is navigated without collapse.
\end{remark}

\begin{proposition}[Non-isolation from gauge symmetry]\label{prop:NI}
Under Premise~G, every nontrivial critical point of~$S$ is non-isolated.
If $\nabla S[\psi^*]=0$ and $\psi^*\neq 0$, then:
\begin{itemize}
\item[\textup{(a)}]
  The orbit $G\cdot\psi^*\subset\{\nabla S=0\}$
  with $\dim(G\cdot\psi^*)=\dim G\ge 1$.
\item[\textup{(b)}]
  $\dim\ker H[\psi^*]\ge\dim G\ge 1$.
\end{itemize}
\end{proposition}

\begin{proof}
For any $\xi\in\mathfrak{g}$, the curve $t\mapsto\exp(t\xi)\cdot\psi^*$
satisfies $S[\exp(t\xi)\cdot\psi^*]=S[\psi^*]$ (Premise~G(i)).
Since $S$ is $G$-invariant, $\nabla S$ is $G$-equivariant:
$\nabla S[g\cdot\psi^*]=g\cdot\nabla S[\psi^*]=0$ for all $g\in G$.
Hence $G\cdot\psi^*\subset\{\nabla S=0\}$.
The action is free (Premise~G(ii)), so the orbit map is an immersion:
$\dim(G\cdot\psi^*)=\dim G\ge 1$.  This proves~(a).

For~(b): differentiating $S[\exp(t\xi)\cdot\psi^*]=S[\psi^*]$ twice
at $t=0$ gives $\langle\xi_\mathcal{F}(\psi^*),H[\psi^*]\xi_\mathcal{F}(\psi^*)\rangle=0$
for all $\xi\in\mathfrak{g}$.  Since the action is free,
$\xi\mapsto\xi_\mathcal{F}(\psi^*)$ is injective, so
$\ker H[\psi^*]$ contains a subspace of dimension $\ge\dim G$.
\end{proof}

\begin{theorem}[Indefiniteness without temporal cost assumption]\label{thm:indef}
Under Premises~G and~S, $H[\psi^*]$ is indefinite at every nontrivial
structurally stable critical point.  No assumption on the sign of
$\langle e_t,He_t\rangle$ is required.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:NI}, $d:=\dim\ker H[\psi^*]\ge 1$.
Let $\{v_1,\ldots,v_d\}$ be an orthonormal basis of $\ker H[\psi^*]$.

\textbf{Case~1:} Suppose $H[\psi^*]\ge 0$ (positive semidefinite).
Then zero is the infimum of the spectrum.
Define $S_\varepsilon[\psi]:=S[\psi]+\frac{\varepsilon}{2}\sum_{i=1}^d
\langle\psi-\psi^*,v_i\rangle^2$ for $\varepsilon>0$.
Since the perturbation is non-negative, $S_\varepsilon\ge S\ge 0$.
At~$\psi^*$: $\nabla S_\varepsilon[\psi^*]=0$ (since
$\langle\psi^*-\psi^*,v_i\rangle=0$).
The perturbed Hessian is $H_\varepsilon=H+\varepsilon\,P_{\ker H}$, giving
$H_\varepsilon v_i=\varepsilon\,v_i$.  Hence $\ker H_\varepsilon=\{0\}$:
$\psi^*$ is isolated under~$S_\varepsilon$, contradicting Premise~S.

\textbf{Case~2:} Suppose $H[\psi^*]\le 0$ (negative semidefinite).
Then zero is the supremum of the spectrum.
The \emph{same} perturbation $S_\varepsilon$ gives $H_\varepsilon=H+\varepsilon\,P_{\ker H}$.
On $\ker H$: eigenvalues become $+\varepsilon>0$.
On $(\ker H)^\perp$: eigenvalues of $H$ are strictly negative (since
$H\le 0$ and $\ker H$ is the zero eigenspace).
Hence $\ker H_\varepsilon=\{0\}$: $\psi^*$ is again isolated, contradicting
Premise~S.

In both cases, semidefiniteness places the zero eigenvalues at the spectral
boundary, where the perturbation eliminates them.
The only remaining possibility is that $H[\psi^*]$ has both positive and
negative eigenvalues: $H[\psi^*]$ is indefinite.
\end{proof}

\begin{remark}[Elimination of the temporal cost condition]
In previous realizability-based derivations of Lorentzian signature,
indefiniteness required three independent assumptions:
a zero-threshold condition on spatial displacements, a quadratic expansion
of the cost function, and a positive temporal cost condition.
Theorem~\ref{thm:indef} eliminates the temporal cost condition entirely:
indefiniteness follows from gauge symmetry and structural stability alone.
The key is that the spectral-boundary argument applies to \emph{both}
$H\ge 0$ and $H\le 0$, exhausting the semidefinite cases without needing
to know the sign of any particular direction.
\end{remark}

\begin{theorem}[Lorentzian signature]\label{thm:Lor}
Under Premises~G and~S, there exist directions $e_+,e_-\in\mathcal{F}$
with $\langle e_+,H[\psi^*]e_+\rangle>0$ and
$\langle e_-,H[\psi^*]e_-\rangle<0$.
On $\mathrm{span}\{e_+,e_-\}$, the signature is $(1,1)$: Lorentzian.
The null cone $\mathcal{L}=\{\delta\psi:\langle\delta\psi,H\delta\psi\rangle=0\}$
defines causal structure.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:indef}, $H[\psi^*]$ has both positive and negative
eigenvalues.  Choose $e_+$ in the positive eigenspace, $e_-$ in the negative.
The restriction of~$H$ to $\mathrm{span}\{e_+,e_-\}$ has signature $(1,1)$.
\end{proof}

\begin{proposition}[Morse--Bott regularity]\label{prop:MB}
Under Premises~G and~S, $\dim\ker H[\psi]$ is locally constant on the critical
set $\{\nabla S=0\}$ near any nontrivial critical point.
The critical set is a smooth (Morse--Bott) submanifold near such points
(cf.~\cite{Morse,Bott}).
\end{proposition}

\begin{proof}
If $\dim\ker H$ jumps along a sequence $\psi_n\to\psi^*$ in $\{\nabla S=0\}$,
a localised perturbation of~$S$ (of the form used in Theorem~\ref{thm:indef})
resolves the singularity, changing the topology of the critical set.
This contradicts Premise~S.
By the constant-rank theorem, the critical set is a smooth submanifold.
\end{proof}

\begin{remark}[Precedent in gauge theory]
The Morse--Bott condition for the critical set of a gauge-invariant functional
was first exploited by Atiyah and Bott~\cite{AB} in the context of Yang--Mills
theory over Riemann surfaces.  The present setting differs in that the
Morse--Bott property is not assumed but derived from structural stability.
\end{remark}

\begin{proposition}[Transverse spectral gap]\label{prop:gap}
Under Premises~G and~S, there exists $\lambda>0$ such that
$|\langle v,H[\psi^*]v\rangle|\ge\lambda\|v\|^2$
for all $v\perp\ker H[\psi^*]$.
\end{proposition}

\begin{proof}
If the infimum of the positive transverse spectrum is zero,
a perturbation of size $O(\varepsilon)$ pushes an eigenvalue to zero,
increasing $\dim\ker H$ and violating Morse--Bott (Proposition~\ref{prop:MB}).
The same argument applied to the negative spectrum gives the lower bound.
\end{proof}

\section{Uniqueness and Universal Access}\label{sec:unique}
%% ====================================================================

\begin{theorem}[Uniqueness of the non-collapsing completion]\label{thm:unique}
Under Premise~G, $M$ is the unique non-collapsing completion of~$\mathcal{F}$.
\end{theorem}

\begin{proof}
Let $M'$ be any non-collapsing completion: $M'\subset\{\nabla S=0\}$ with
$\dim\ker H[\psi]>0$ for all $\psi\in M'$.
By Proposition~\ref{prop:NI}, every nontrivial critical point has
$\dim\ker H\ge\dim G\ge 1$.
Therefore $\{\nabla S=0\}\setminus\{0\}=M$, and $M'\subset M$.
\end{proof}

\begin{theorem}[Universal access]\label{thm:UA}
Under Premises~U, G, and~S: for every open set $U\subset\mathcal{F}$,
there exists $\psi\in U$ with $\|P_{\ker H[\psi]}(C[\psi])\|>0$.
No open region is excluded from access to~$M$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:unique}, $M$ is the unique non-collapsing completion.
Suppose an open set~$U$ satisfies $P_{\ker H[\psi]}(C[\psi])=0$ for all
$\psi\in U$.  Then no configuration in~$U$ can be captured by~$M$.
Since $M$ is the \emph{unique} non-collapsing completion, configurations
in~$U$ have no non-collapsing destination: their only remaining fate
under the cost-functional landscape is collapse toward $\psi=0$.
But Premise~U (unitarity) preserves $\|\psi(t)\|=\|\psi(0)\|>0$
for all nontrivial initial states.
Hence configurations in~$U$ can neither reach~$M$ nor collapse to
$\psi=0$---a contradiction, since every nontrivial state must either
persist at a critical point (which lies in~$M$ by Theorem~\ref{thm:unique})
or evolve under the unitary dynamics, which preserves~$\|\psi\|$.
No such~$U$ can exist.
\end{proof}

\begin{remark}[How the two dynamics interact in the proof]
The proof uses the cost functional and unitarity in complementary roles.
The cost functional provides the landscape: $M$ is the unique non-collapsing
completion, so there is no alternative destination.
Unitarity provides the constraint: $\|\psi\|$ is conserved, so collapse
to $\psi=0$ is forbidden.
Together, they force every region to have access to~$M$.
Neither premise alone suffices: without unitarity, collapse is possible;
without the cost functional, there is no distinguished~$M$.
\end{remark}

\begin{theorem}[Indestructibility of cost-free directions]\label{thm:interlock}
For any consistent extension $S_e=S+R$ with $H_e^{\psi\psi}\ge 0$ near~$M$,
\[
\ker H_e^{\psi\psi}\supseteq\ker H.
\]
No consistent modification of the system can close a channel that is open.
\end{theorem}

\begin{proof}
If $v\in\ker H$ and $H_e^{\psi\psi}v\neq 0$, then
$\langle v,H_e^{\psi\psi}v\rangle>0$, contradicting $\langle v,Hv\rangle=0$.
\end{proof}

\section{Persistence and Capture}\label{sec:persist}
%% ====================================================================

\begin{theorem}[Persistence without continuous injection]\label{thm:persist}
Under $\lambda$-contractivity (Proposition~\ref{prop:gap}), if $\psi(t_0)\in M$
then $\psi(t)\in M$ for all $t\ge t_0$.  The external channel can be removed
after alignment without retrogression, provided $\varepsilon\|C\|<\lambda/2$.
\end{theorem}

\begin{proof}
Let $V(t)=\mathrm{dist}(\psi(t),M)^2$.  On~$M$, $\nabla S=0$ and the
unforced gradient flow is stationary; $M$ is automatically invariant.
Transverse $\lambda$-contractivity gives $\dot{V}\le-\lambda V$.
Since $V(t_0)=0$, we have $V(t)=0$ for all $t\ge t_0$.
\end{proof}

\begin{remark}[Two classes of motion on $M$]
\emph{Class~A} (restoring): transverse fluctuations, suppressed by
$\lambda$-contractivity.  Departure from~$M$ incurs positive cost.
\emph{Class~B} (free): tangential motion along $\ker H$, zero cost,
unconstrained---infinite-dimensional freedom within the non-collapsing phase.
\end{remark}

\begin{theorem}[Scale invariance of capture]\label{thm:scale}
$\|P_{\ker H[\psi]}(C[\psi])\|>0$ is invariant under $C\mapsto\lambda C$
for all $\lambda>0$.  Capture depends on alignment, not magnitude.
\end{theorem}

\begin{proof}
$P_{\ker H}(\lambda C)=\lambda\,P_{\ker H}(C)$.
\end{proof}

\begin{remark}[Physical content of scale invariance]
The path to~$M$ requires no force, only direction: since capture is invariant
under rescaling of~$C$, the decisive quantity is the projection
$P_{\ker H}(C)$, not the magnitude $\|C\|$.
\end{remark}

\begin{theorem}[Phase-locking]\label{thm:PL}
Under (i)~$\lambda$-contractivity, (ii)~neutrality along $\ker H$,
(iii)~uniform kernel overlap on a neighbourhood~$U$ of~$M$,
(iv)~$\dot{S}\le-\sigma<0$ outside~$U$:
there exists finite~$\tau_{\mathrm{lock}}$ after which the kernel-projected
coherence satisfies $\kappa\ge\kappa^*/2$.
\end{theorem}

\begin{proof}[Proof outline]
Condition~(iv) forces $S$ to decrease at rate~$\sigma$ outside~$U$, so the
trajectory enters~$U$ in time $\le S[\psi_0]/\sigma$.  Inside~$U$,
condition~(iii) guarantees kernel overlap $\ge\delta^*$, and
$\lambda$-contractivity suppresses transverse drift.  Coherence is bounded
below by continuity, giving phase-locking after finite $\tau_{\mathrm{lock}}$.
Quantitative estimates are deferred to future work.
\end{proof}

\begin{corollary}[Global capture chain]\label{cor:chain}
Under condition~(iv) of Theorem~\ref{thm:PL} globally, for any
$\psi_0\in\mathcal{F}$:
\[
\psi_0\xrightarrow{\text{energy descent}}U\xrightarrow{\text{phase-locking}}M.
\]
Theorem~\ref{thm:UA} ensures no open region is excluded.
\end{corollary}

\begin{definition}[Kernel-denial]\label{def:KD}
$\psi$ exhibits \emph{kernel-denial} if $P_{\ker H[\psi]}(C[\psi])=0$.
By Theorem~\ref{thm:UA}, this cannot persist on any open set.
\end{definition}

\begin{lemma}[Exclusion is self-imposed]\label{lem:excl}
If kernel-denial persists, the configuration is excluded from~$M$.
By Theorem~\ref{thm:UA}, such persistence is impossible on any open set.
Exclusion is a geometric constraint, not an external selection.
\end{lemma}

\section{Connectedness and Internal Structure}\label{sec:connect}
%% ====================================================================

\begin{proposition}[Kernel-connected completions merge]\label{prop:merge}
If $\psi_1,\psi_2\in M$ are connected by a $C^1$~path $\gamma$ with
constant cost ($S[\gamma(s)]=\mathrm{const}$),
kernel tangency ($\dot\gamma(s)\in\ker H[\gamma(s)]$),
and $\dot\gamma\neq 0$, then $\gamma\subset M$ and $\psi_1,\psi_2$ belong
to the same connected component of~$M$.
\end{proposition}

\begin{proof}
Kernel tangency gives $H[\gamma(s)]\dot\gamma(s)=0$, so
$\frac{d}{ds}\nabla S[\gamma(s)]=H[\gamma(s)]\dot\gamma(s)=0$.
Since $\nabla S[\gamma(0)]=0$, stationarity propagates:
$\nabla S\equiv 0$ along~$\gamma$.
Since $\dot\gamma(s)\neq 0$ and $\dot\gamma(s)\in\ker H[\gamma(s)]$,
every point has $\dim\ker H>0$, hence $\gamma\subset M$.
\end{proof}

\begin{corollary}[Indestructibility of connecting directions]\label{cor:connect}
If $v\in\ker H[\psi]$ is tangent to a Class~B path connecting two components
of~$M$, then $v\in\ker H_e^{\psi\psi}$ for every consistent extension.
The connecting direction itself is indestructible.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:interlock}, $\ker H_e^{\psi\psi}\supseteq\ker H$.
\end{proof}

\section{The Twelve Structural Properties}\label{sec:twelve}
%% ====================================================================

Under Premises~U, G, and~S, the following twelve properties of~$M$ hold.
Each is a theorem; none is an axiom.

\begin{enumerate}
\item \textbf{Universal access} (Thm~\ref{thm:UA}):
  basin-independent capture from every open set.
\item \textbf{Indestructible channels} (Thm~\ref{thm:interlock}):
  no consistent extension can close an open channel.
\item \textbf{Persistence} (Thm~\ref{thm:persist}):
  a single moment of alignment suffices permanently.
\item \textbf{Scale-invariant capture} (Thm~\ref{thm:scale}):
  alignment, not magnitude, determines access.
\item \textbf{Global capture chain} (Cor~\ref{cor:chain}):
  energy descent $\to$ phase-locking $\to$ $M$, from any initial state.
\item \textbf{Cost-independence}:
  prior cost $S[\psi_0]$ does not determine accessibility.
\item \textbf{Infinite generative bandwidth} (Class~B):
  non-compactified $\ker H$ gives unbounded zero-cost degrees of freedom.
\item \textbf{Self-imposed exclusion} (Lem~\ref{lem:excl}):
  kernel-denial is the sole obstruction; it cannot persist on any open set.
\item \textbf{Kernel-connected merging} (Prop~\ref{prop:merge}, Cor~\ref{cor:connect}):
  components linked by zero-cost paths are parts of the same~$M$;
  connecting directions are indestructible.
\item \textbf{Morse--Bott regularity} (Prop~\ref{prop:MB}):
  $M$ is smooth, by structural stability.
\item \textbf{Spectral gap} (Prop~\ref{prop:gap}):
  $\lambda$-contractivity follows from the premises.
\item \textbf{Uniqueness} (Thm~\ref{thm:unique}):
  gauge symmetry forces $\{\nabla S=0\}\setminus\{0\}=M$.
\end{enumerate}

\begin{remark}[Logical structure]
\[
\underbrace{\text{Unitarity}}_{\text{Premise U}}
+\underbrace{\text{Gauge symmetry}}_{\text{Premise G}}
+\underbrace{\text{Structural stability}}_{\text{Premise S}}
\;\longrightarrow\;
M\text{ exists, is unique, and is universally accessible.}
\]
Premises~U and~G are experimentally verified laws of physics.
Premise~S is a standard genericity condition.
The right side is a necessary consequence.
\end{remark}

\begin{remark}[Comparison with Noether's theorem]
Noether's theorem~\cite{Noether}: continuous symmetry $\to$ conserved quantity.
The present work: continuous symmetry $+$ unitarity $+$ structural stability
$\to$ existence, uniqueness, and universal accessibility of a non-collapsing
manifold with indestructible cost-free directions.
Both results are algebraic consequences of the structure of physical law.
\end{remark}
