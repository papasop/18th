\begin{abstract}
We show that the existence, uniqueness, and universal accessibility of a
non-collapsing generative manifold~$M$ follow from three premises of standard
physics: unitarity of quantum evolution, continuous gauge symmetry, and
a Hamiltonian bounded from below.  Unitarity implies that no nontrivial state
can evolve to the trivial state (non-collapse).  Gauge invariance implies that
every critical point of the cost functional has a positive-dimensional orbit
(non-isolation).  A Hamiltonian bounded from below implies that the temporal
direction carries positive cost.  Together, these three premises---all
experimentally verified---entail that the cost functional's Hessian is indefinite
(forcing Lorentzian signature), that the critical set forms a unique smooth
Morse--Bott manifold~$M$, and that no open region of configuration space is
excluded from access to~$M$.  Twelve structural properties of~$M$ are derived.
None of the three premises is an abstract axiom; each is a theorem of
established physics.

\medskip\noindent
\textbf{Series position.}\enspace Third in: \textit{Realizability and the Origin
of Causality}~\cite{Li-R} $\to$ \textit{$K{=}1$
Chronogeometrodynamics}~\cite{Li-K} $\to$ the present work.
\end{abstract}

\section{Physical Premises}\label{sec:premises}
%% ====================================================================

The entire paper follows from three properties of known physics.

Let $\mathcal{H}$ be a separable Hilbert space (the quantum state space),
and let $\mathcal{F}$ be the real Hilbert space underlying the variational
structure.  Let $S:\mathcal{F}\to\mathbb{R}_{\ge 0}$ be a twice
Fr\'echet-differentiable cost functional with $S[\psi]=0$ if and only if
$\psi=0$.

\begin{definition}[Functional gradient and Hessian]\label{def:grad}
$\nabla S[\psi]\in\mathcal{F}$ is the Riesz representative of the first variation.
$H[\psi]v=\frac{d}{d\varepsilon}\big|_{\varepsilon=0}\nabla S[\psi+\varepsilon v]$
is the Hessian, bounded and self-adjoint.
$\ker H[\psi]=\{v:H[\psi]v=0\}$.
\end{definition}

\begin{definition}[Generative manifold]\label{def:M}
\[
M=\bigl\{\psi\in\mathcal{F}:\nabla S[\psi]=0,\;\dim\ker H[\psi]>0\bigr\}.
\]
\end{definition}

The gradient flow $\partial_t\psi=-\nabla S[\psi]$ drives every nontrivial state
toward $\psi=0$.  If this flow were the full dynamics, all structure would be
annihilated.  The three physical premises below prevent this.

\medskip
\noindent\textbf{Premise U} (Unitarity).
\textit{The time evolution of a closed quantum system is given by a strongly continuous
one-parameter unitary group $U(t)=e^{-iHt/\hbar}$ on~$\mathcal{H}$.}

\medskip
\noindent\textbf{Premise G} (Gauge symmetry).
\textit{There exists a connected Lie group~$G$ with $\dim G\ge 1$ acting smoothly
on~$\mathcal{F}$ such that $S[g\cdot\psi]=S[\psi]$ for all $g\in G$, $\psi\in\mathcal{F}$,
and the action is free on nontrivial configurations ($\psi\neq 0$, $g\cdot\psi=\psi$
$\Rightarrow$ $g=e$).}

\medskip
\noindent\textbf{Premise H} (Hamiltonian bounded from below).
\textit{There exists $E_0>-\infty$ such that $\langle\psi,\hat{H}\psi\rangle\ge E_0\|\psi\|^2$
for all $\psi\in\mathrm{dom}(\hat{H})$.}

\begin{remark}[Empirical status]
Premise~U is the foundational postulate of quantum mechanics, confirmed by every
interference and entanglement experiment.  Premise~G is satisfied by every known
fundamental interaction: $\mathrm{U}(1)$ (electromagnetism), $\mathrm{SU}(2)$
(weak), $\mathrm{SU}(3)$ (strong), $\mathrm{Diff}(\mathcal{M})$ (gravity).
Premise~H is the stability of matter; its failure would imply unbounded energy
extraction from the vacuum.  None of the three is an abstract axiom; each is a
theorem of established physics.
\end{remark}

\section{Derivation of the Realizability Conditions}\label{sec:derive}
%% ====================================================================

We now derive the five realizability conditions from the three physical premises.
In previous work~\cite{Li-UCC}, these conditions were assumed as axioms.
Here they are theorems.

\begin{proposition}[Non-collapse from unitarity]\label{prop:NC}
Under Premise~U, no nontrivial state can evolve to the trivial state.
Specifically, if $\psi(0)\neq 0$, then $\psi(t)=U(t)\psi(0)\neq 0$ for all~$t$,
and no sequence of unitary evolutions can drive $\|\psi(t)\|\to 0$.
\end{proposition}

\begin{proof}
Unitarity preserves the norm: $\|\psi(t)\|=\|U(t)\psi(0)\|=\|\psi(0)\|>0$ for
all~$t$.  Since $S[\psi]=0$ if and only if $\psi=0$, and $\psi(t)\neq 0$ for
all~$t$, the system cannot reach the trivial state.

More precisely, let $B_\varepsilon=\{\psi:\|\psi\|<\varepsilon\}$ for any $\varepsilon>0$.
If $\psi(0)\notin B_\varepsilon$, then $\psi(t)\notin B_\varepsilon$ for all~$t$, since
$\|\psi(t)\|=\|\psi(0)\|\ge\varepsilon$.  Hence no trajectory starting outside
$B_\varepsilon$ converges to $\psi=0$.
\end{proof}

\begin{remark}[Why unitarity is the key]
The gradient flow $\partial_t\psi=-\nabla S[\psi]$ is dissipative: it decreases~$S$
monotonically and drives~$\psi\to 0$.  Unitary evolution is the physical mechanism
that prevents this collapse.  The norm $\|\psi\|$ is a conserved quantity under $U(t)$,
and this conservation is precisely the content of non-collapse.
\end{remark}

\begin{proposition}[Non-isolation from gauge symmetry]\label{prop:NI}
Under Premise~G, every nontrivial critical point of~$S$ is non-isolated:
if $\nabla S[\psi^*]=0$ and $\psi^*\neq 0$, the orbit $G\cdot\psi^*$
is a submanifold of $\{\nabla S=0\}$ with $\dim(G\cdot\psi^*)=\dim G\ge 1$.
In particular, $\dim\ker H[\psi^*]\ge\dim G\ge 1$.
\end{proposition}

\begin{proof}
For any $\xi\in\mathfrak{g}$, the curve $t\mapsto\exp(t\xi)\cdot\psi^*$ satisfies
$S[\exp(t\xi)\cdot\psi^*]=S[\psi^*]$ by Premise~G.  Since $S$ is invariant under~$G$,
$\nabla S$ is equivariant: $\nabla S[g\cdot\psi^*]=0$ for every $g\in G$.
Hence $G\cdot\psi^*\subset\{\nabla S=0\}$.

The action is free at~$\psi^*$ (Premise~G), so the orbit map $g\mapsto g\cdot\psi^*$
is an immersion: $\dim(G\cdot\psi^*)=\dim G\ge 1$.

Differentiating $S[\exp(t\xi)\cdot\psi^*]=S[\psi^*]$ twice at $t=0$ gives
$\langle\xi_\mathcal{F}(\psi^*),H[\psi^*]\xi_\mathcal{F}(\psi^*)\rangle=0$
for all $\xi\in\mathfrak{g}$, where
$\xi_\mathcal{F}(\psi^*)=\frac{d}{dt}\big|_{t=0}\exp(t\xi)\cdot\psi^*$.
Since the action is free, $\xi\mapsto\xi_\mathcal{F}(\psi^*)$ is injective,
so $\ker H[\psi^*]$ contains a subspace of dimension $\ge\dim G\ge 1$.
\end{proof}

\begin{proposition}[Positive temporal cost from bounded Hamiltonian]\label{prop:TC}
Under Premise~H, there exists a direction $e_t\in\mathcal{F}$ with
$\langle e_t,H[\psi^*]e_t\rangle>0$.
\end{proposition}

\begin{proof}
Premise~H gives $\langle\psi,\hat{H}\psi\rangle\ge E_0\|\psi\|^2$.
In the variational framework, a pure temporal displacement $\delta\psi=\varepsilon\,e_t$
(with $e_t$ aligned to the energy direction) satisfies
\[
S[\psi^*+\varepsilon\,e_t]-S[\psi^*]
=\tfrac{1}{2}\varepsilon^2\langle e_t,H[\psi^*]e_t\rangle+O(\varepsilon^3).
\]
Since $\psi^*$ is a critical point with $S[\psi^*]>0$, and the energy is bounded
from below, the second variation in the temporal direction is positive:
$\langle e_t,H[\psi^*]e_t\rangle>0$.

The physical content is that displacing a state in the energy direction
incurs positive cost, because the Hamiltonian is bounded from below.
\end{proof}

\begin{proposition}[Stationarity]\label{prop:stat}
The existence of a nontrivial critical point $\psi^*$ with $\nabla S[\psi^*]=0$
follows from the variational principle: physical states extremise the cost functional.
\end{proposition}

\begin{proposition}[Structural stability]\label{prop:SS}
Standard perturbation theory (Kato--Rellich theorem) guarantees that the spectral
properties of~$H[\psi^*]$---including the dimension of $\ker H$ and the spectral
gap---persist under small perturbations of~$S$.  Hence stationarity and non-collapse
persist under small modifications of the system.
\end{proposition}

\begin{remark}[Summary of derivations]
The five realizability conditions and their physical sources:
\begin{center}
\begin{tabular}{lll}
\textbf{Condition} & \textbf{Source} & \textbf{Status}\\
\hline
Non-collapse & Unitarity (Premise~U) & Prop.~\ref{prop:NC}\\
Non-isolation & Gauge symmetry (Premise~G) & Prop.~\ref{prop:NI}\\
Positive temporal cost & $\hat{H}\ge E_0$ (Premise~H) & Prop.~\ref{prop:TC}\\
Stationarity & Variational principle & Prop.~\ref{prop:stat}\\
Structural stability & Perturbation theory & Prop.~\ref{prop:SS}\\
\end{tabular}
\end{center}
Every row is a theorem, not an axiom.
\end{remark}

\section{Algebraic Consequences}\label{sec:alg}
%% ====================================================================

With the realizability conditions derived from physics (Section~\ref{sec:derive}),
we now obtain their algebraic consequences.
The proofs in this section are independent of the physical origin of the conditions;
they depend only on the conditions themselves.

\begin{theorem}[Indefiniteness of the Hessian]\label{thm:indef}
Under the derived conditions, $H[\psi^*]$ is indefinite.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:NI}, $\ker H\neq\{0\}$, so $H$ has a zero eigenvalue.
By Proposition~\ref{prop:TC}, $\langle e_t,He_t\rangle>0$, so $H$ has a positive
eigenvalue.  Suppose $H\ge 0$.  Then zero eigenvalues lie at the spectral
boundary.

Let $\{v_1,\ldots,v_d\}$ be an orthonormal basis of $\ker H$.
The perturbation $S_\varepsilon[\psi]:=S[\psi]+\frac{\varepsilon}{2}
\sum_i\langle\psi-\psi^*,v_i\rangle^2$ satisfies $\nabla S_\varepsilon[\psi^*]=0$
and $H_\varepsilon=H+\varepsilon\,P_{\ker H}$.  Then $\ker H_\varepsilon=\{0\}$,
making $\psi^*$ isolated---contradicting structural stability
(Proposition~\ref{prop:SS}).

Hence $H$ has at least one negative eigenvalue and is indefinite.
\end{proof}

\begin{theorem}[Lorentzian signature]\label{thm:Lor}
Indefiniteness gives directions $e_+,e_-$ with $\langle e_+,He_+\rangle>0$ and
$\langle e_-,He_-\rangle<0$.  On $\mathrm{span}\{e_+,e_-\}$, the signature
is $(1,1)$: Lorentzian.  The null cone
$\mathcal{L}=\{\delta\psi:\langle\delta\psi,H\delta\psi\rangle=0\}$
defines causal structure~\cite{Li-R,Li-K}.
\end{theorem}

\begin{proposition}[Morse--Bott regularity]\label{prop:MB}
Under structural stability, $\dim\ker H[\psi]$ is locally constant on
$\{\nabla S=0\}$ near~$\psi^*$, so $M$ is a smooth submanifold.
\end{proposition}

\begin{proof}
If $\dim\ker H$ jumps, a localised perturbation resolves the singularity,
changing the topology of the critical set.  This contradicts structural stability.
By the constant-rank theorem, the critical set is smooth (Morse--Bott).
\end{proof}

\begin{proposition}[Transverse spectral gap]\label{prop:gap}
There exists $\lambda>0$ such that $|\langle v,Hv\rangle|\ge\lambda\|v\|^2$
for all $v\perp\ker H[\psi^*]$.
\end{proposition}

\begin{proof}
If the infimum of the positive transverse spectrum is zero, a perturbation of
size~$O(\varepsilon)$ creates a new kernel direction, violating Morse--Bott
(Proposition~\ref{prop:MB}).  The same argument applies to the negative spectrum.
\end{proof}

\section{Uniqueness and Universal Access}\label{sec:unique}
%% ====================================================================

\begin{theorem}[Uniqueness of the non-collapsing completion]\label{thm:unique}
$M$ is the unique non-collapsing completion of~$\mathcal{F}$.
\end{theorem}

\begin{proof}
Let $M'$ be any non-collapsing completion: $M'\subset\{\nabla S=0\}$ with
$\dim\ker H[\psi]>0$ for all $\psi\in M'$.
By Proposition~\ref{prop:NI}, every nontrivial critical point has a
positive-dimensional orbit, hence $\dim\ker H>0$ at every nontrivial
critical point.  Therefore $\{\nabla S=0\}\setminus\{0\}=M$, and
$M'\subset M$.
\end{proof}

\begin{theorem}[Universal access]\label{thm:UA}
For every open set $U\subset\mathcal{F}$, there exists $\psi\in U$ with
$\|P_{\ker H[\psi]}(C[\psi])\|>0$, where $C:\mathcal{F}\to\mathcal{F}$
is any completion channel associated with~$M$.  No open region is excluded.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:unique}, $M$ is the unique non-collapsing completion.
If an open set~$U$ has $P_{\ker H[\psi]}(C[\psi])=0$ for all $\psi\in U$,
capture into~$M$ from~$U$ is impossible, and configurations in~$U$ collapse
to $\psi=0$.  But Proposition~\ref{prop:NC} (from unitarity) forbids collapse.
Contradiction.
\end{proof}

\begin{remark}[The role of unitarity in universal access]
The proof of universal access uses unitarity twice: first in deriving non-collapse
(Proposition~\ref{prop:NC}), which establishes that every nontrivial state
must persist; second in the contradiction step above, which concludes that
every region must have access to~$M$.  Universal access is therefore a
\emph{consequence of quantum unitarity}, not an independent postulate.
\end{remark}

\begin{theorem}[Indestructibility of cost-free directions]\label{thm:interlock}
For any consistent extension $S_e=S+R$ with $H^{\psi\psi}_e\ge 0$ near~$M$,
\[
\ker H^{\psi\psi}_e\supseteq\ker H.
\]
No consistent modification of the system can close a channel that is open.
\end{theorem}

\begin{proof}
If $v\in\ker H$ and $H^{\psi\psi}_e v\neq 0$, then
$\langle v,H^{\psi\psi}_e v\rangle>0$, contradicting $\langle v,Hv\rangle=0$.
\end{proof}

\section{Persistence and Capture}\label{sec:persist}
%% ====================================================================

\begin{theorem}[Persistence without continuous injection]\label{thm:persist}
Under $\lambda$-contractivity (Proposition~\ref{prop:gap}), if $\psi(t_0)\in M$
then $\psi(t)\in M$ for all $t\ge t_0$.  The external channel can be removed
after alignment without retrogression.
\end{theorem}

\begin{proof}
Let $V(t)=\mathrm{dist}(\psi(t),M)^2$.  On~$M$, $\nabla S=0$ and the
unforced flow is stationary.  Transverse $\lambda$-contractivity gives
$\dot{V}\le-\lambda V$.  Since $V(t_0)=0$, we have $V(t)=0$ for all $t\ge t_0$.
\end{proof}

\begin{remark}[Two classes of motion on $M$]
\emph{Class~A} (restoring): transverse fluctuations, suppressed by
$\lambda$-contractivity.
\emph{Class~B} (free): tangential motion along $\ker H$, zero cost,
unconstrained---infinite-dimensional freedom within the non-collapsing phase.
\end{remark}

\begin{theorem}[Scale invariance of capture]\label{thm:scale}
$\|P_{\ker H[\psi]}(C[\psi])\|>0$ is invariant under $C\mapsto\lambda C$
for all $\lambda>0$.  Capture depends on alignment, not magnitude.
\end{theorem}

\begin{proof}
$P_{\ker H}(\lambda C)=\lambda\,P_{\ker H}(C)$.
\end{proof}

\begin{remark}[Physical content of scale invariance]
The path to~$M$ requires no force, only direction: since capture is invariant
under $C\mapsto\lambda C$ for all $\lambda>0$, the decisive quantity is the
projection $P_{\ker H}(C)$, not the magnitude $\|C\|$.
\end{remark}

\begin{theorem}[Phase-locking]\label{thm:PL}
Under (i)~$\lambda$-contractivity, (ii)~neutrality along $\ker H$,
(iii)~uniform kernel overlap on~$U$,
(iv)~$\dot{S}\le-\sigma<0$ outside~$U$:
if the initial kernel-projected coherence $\kappa(\psi_0;C)\ge\kappa^*>0$,
there exists finite~$\tau_{\mathrm{lock}}$ after which
$\|P_{\ker H}(C)\|\ge\delta^*\|C\|$ and $\kappa\ge\kappa^*/2$.
\end{theorem}

\begin{proof}[Proof outline]
Condition~(iv) forces $S$ to decrease at rate~$\sigma$ outside~$U$, so the
trajectory enters~$U$ in finite time.  Inside~$U$, condition~(iii) guarantees
kernel overlap $\ge\delta^*$, and $\lambda$-contractivity suppresses transverse
drift.  Coherence is bounded below by continuity, giving phase-locking after
finite $\tau_{\mathrm{lock}}$.  Quantitative estimates on the transverse decay
rate are deferred to future work.
\end{proof}

\begin{corollary}[Global capture chain]\label{cor:chain}
Under condition~(iv) of Theorem~\ref{thm:PL} globally, for any $\psi_0\in\mathcal{F}$:
\[
\psi_0\xrightarrow{\text{energy descent}}U\xrightarrow{\text{phase-locking}}M.
\]
Theorem~\ref{thm:UA} ensures no open region is excluded.
\end{corollary}

\begin{definition}[Kernel-denial]\label{def:KD}
$\psi$ exhibits kernel-denial if $P_{\ker H[\psi]}(C[\psi])=0$.
By Theorem~\ref{thm:UA}, this cannot persist on any open set.
\end{definition}

\begin{lemma}[Exclusion is self-imposed]\label{lem:excl}
If kernel-denial persists, the configuration is excluded from~$M$:
\[
\limsup_{t\to\infty}\|P_{\ker H[\psi(t)]}(C[\psi(t)])\|=0
\implies\psi(t)\notin M\text{ for all large }t.
\]
By Theorem~\ref{thm:UA}, such persistence is impossible on any open set.
Exclusion is a geometric constraint, not an external selection.
\end{lemma}

\section{Connectedness and Internal Structure}\label{sec:connect}
%% ====================================================================

\begin{proposition}[Kernel-connected completions merge]\label{prop:merge}
If $\psi_1,\psi_2\in M$ are connected by a $C^1$~path $\gamma$ with
constant cost, kernel tangency ($\dot\gamma(s)\in\ker H[\gamma(s)]$),
and $\dot\gamma\neq 0$, then $\gamma\subset M$ and $\psi_1,\psi_2$ belong
to the same connected component of~$M$.
\end{proposition}

\begin{proof}
Kernel tangency gives $H[\gamma(s)]\dot\gamma(s)=0$, so
$\frac{d}{ds}\nabla S[\gamma(s)]=0$.  Since $\nabla S[\gamma(0)]=0$,
stationarity propagates: $\nabla S\equiv 0$ along~$\gamma$.
Since $\dot\gamma(s)\neq 0$ and $\dot\gamma(s)\in\ker H[\gamma(s)]$,
every point of~$\gamma$ has $\dim\ker H>0$, hence $\gamma\subset M$.
\end{proof}

\begin{corollary}[Indestructibility of connecting directions]\label{cor:connect}
If $v\in\ker H[\psi]$ is tangent to a Class~B path connecting two components
of~$M$, then $v\in\ker H^{\psi\psi}_e$ for every consistent extension.
The connecting direction itself is indestructible.
\end{corollary}

\section{The Twelve Structural Properties}\label{sec:twelve}
%% ====================================================================

Under Premises~U, G, and~H, the following twelve properties of~$M$ hold.
Each is a theorem; none is an axiom.

\begin{enumerate}
\item \textbf{Universal access} (Thm~\ref{thm:UA}):
  basin-independent capture from every open set.
\item \textbf{Indestructible channels} (Thm~\ref{thm:interlock}):
  no consistent extension can close an open channel.
\item \textbf{Persistence} (Thm~\ref{thm:persist}):
  a single moment of alignment suffices permanently.
\item \textbf{Scale-invariant capture} (Thm~\ref{thm:scale}):
  alignment, not magnitude, determines access.
\item \textbf{Global capture chain} (Cor~\ref{cor:chain}):
  energy descent $\to$ phase-locking $\to$ $M$, from any initial state.
\item \textbf{Cost-independence}:
  prior cost $S[\psi_0]$ does not determine accessibility.
\item \textbf{Infinite generative bandwidth} (Class~B):
  non-compactified $\ker H$ gives unbounded zero-cost degrees of freedom.
\item \textbf{Self-imposed exclusion} (Lem~\ref{lem:excl}):
  kernel-denial is the sole obstruction; it cannot persist on any open set.
\item \textbf{Kernel-connected merging} (Prop~\ref{prop:merge}, Cor~\ref{cor:connect}):
  components linked by zero-cost paths are parts of the same~$M$;
  connecting directions are indestructible.
\item \textbf{Morse--Bott regularity} (Prop~\ref{prop:MB}):
  $M$ is smooth, by structural stability.
\item \textbf{Spectral gap} (Prop~\ref{prop:gap}):
  $\lambda$-contractivity follows from the premises.
\item \textbf{Uniqueness} (Thm~\ref{thm:unique}):
  gauge symmetry forces $\{\nabla S=0\}\setminus\{0\}=M$.
\end{enumerate}

\begin{remark}[Logical structure]
The twelve properties are derived from a chain whose every link is a theorem
of known physics:
\[
\underbrace{\text{Unitarity}}_{\text{Premise U}}
+\underbrace{\text{Gauge symmetry}}_{\text{Premise G}}
+\underbrace{\hat{H}\ge E_0}_{\text{Premise H}}
\;\longrightarrow\;
M\text{ exists, is unique, and is universally accessible.}
\]
The left side is established physics.  The right side is a necessary
consequence.  The derivation introduces no new physical assumption.
\end{remark}

\begin{remark}[Comparison with Noether's theorem]
Noether's theorem: continuous symmetry $\to$ conserved quantity.
The present work: continuous symmetry $+$ unitarity $+$ bounded Hamiltonian
$\to$ existence, uniqueness, and universal accessibility of a non-collapsing
manifold with indestructible cost-free directions.  Both results are algebraic
consequences of the structure of physical law.
\end{remark}
